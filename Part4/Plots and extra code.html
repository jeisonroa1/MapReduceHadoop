# Plotting funtion
def plotChart(x1,y1,x2,y2,label_1,label_2,title="",x_label="",y_label=""):
    plt.figure(figsize=(12, 4))
    plt.title(title)
    plt.locator_params(axis='x', nbins=10)
    plt.xlabel(x_label)
    plt.ylabel(y_label)
    plt.plot(
        x1, y1, 'o-', label=label_1)
    plt.plot(x2, y2, 'o-', label=label_2)
    plt.legend(loc="best")
